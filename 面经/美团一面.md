- 包装同步函数为异步

```
function packAsync(fn){
	if(typeof fn!=="function"){
		throw ("need function");
	}
	return function(){
		let ctx=this;
		return new Promise((resolve, reject)=>{
		try{
			resolve(fn.call(ctx, ...arguments))
		}catch(e){
			reject(e);
		}
		})
	}
}
```
